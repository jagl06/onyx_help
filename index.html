#!/usr/bin/env python3 # -*- coding: utf-8 -*- """ build_index.py — Construye el indice de busqueda (BM25) para IFAS. Lee los PDFs de las carpetas configuradas (OneDrive sincronizado), extrae el texto pagina por pagina, lo parte en fragmentos con metadata (categoria, documento, pagina) y escribe un unico archivo 'ifas_index.json' que subes al repo 'ifas_help' (junto a index.html). La busqueda es lexica (BM25) y ocurre en el navegador: NO usa embeddings, NO necesita API key, NO tiene limites de tasa y corre en segundos. Requisitos: pip install -U pypdf cryptography Uso: python build_index.py (genera el indice) python build_index.py --dry (solo diagnostico: cuenta y categorias) """ import os import re import json import sys from pypdf import PdfReader # ─────────────────────────── CONFIG ─────────────────────────── CHUNK_CHARS = 1800 # ~tamano de cada fragmento CHUNK_OVERLAP = 320 # solapamiento entre fragmentos OUT_FILE = "ifas_index.json" # Tope de paginas por documento (0 = sin tope). Recorta los manuales gigantes # para que el indice no se vuelva enorme. Los datasheets (cortos) no se afectan. MAX_PAGES_PER_DOC = 80 # Ajusta esta base a tu ruta local de OneDrive BASE = r"C:\Users\Jesus\OneDrive - ISELEC" # (carpeta_raiz, categoria_fija) -> si categoria_fija es None, se usa la # primera subcarpeta debajo de la raiz como categoria. SOURCES = [ (os.path.join(BASE, r"PRODUCTOS\NOTIFIER\DOCUMENTACION NOTIFIER ONYX UNIVERSITY"), None), (os.path.join(BASE, r"PRODUCTOS\NOTIFIER\PRESENTACIONES"), "PRESENTACIONES"), # NFPA 72 excluido del indice publico por derechos de autor (las reglas # generales de diseno viven directamente en la REGLA 7 del system prompt). # (os.path.join(BASE, r"NFPA\NFPA 72 curso"), "NFPA72"), ] # Documentos a excluir (consolidados gigantes, duplicados o con derechos de autor). # Substring del nombre de archivo, en mayusculas. # ALL_DETECTORS duplica las fichas individuales de los detectores (que si se indexan). # MP72/NFPA 72 se excluyen por derechos de autor (ver REGLA 7 del system prompt). EXCLUDE = ["ALL_DOCS_NOTIFIER", "PRESENTACION CONSOLIDADA", "ALL_DETECTORS", "MP72", "NFPA 72", "NFPA72", "NFPA-72"] # ─────────────────────── EXTRACCION / CHUNKING ─────────────────────── def category_for(root, path, fixed): if fixed: return fixed root_name = os.path.basename(root.rstrip("\\/")).lower() parts = os.path.relpath(path, root).split(os.sep)[:-1] # carpetas, sin el archivo # saltar carpetas anidadas que repiten el nombre de la raiz (estructura duplicada) while parts and parts[0].lower() == root_name: parts.pop(0) return parts[0] if parts else "GENERAL" def clean(text): text = text.replace("\x00", " ") text = re.sub(r"[ \t]+", " ", text) text = re.sub(r"\n{3,}", "\n\n", text) return text.strip() def chunk_page(text): """Parte el texto de una pagina en fragmentos con solapamiento.""" text = clean(text) if len(text) <= CHUNK_CHARS: return [text] if text else [] chunks, start = [], 0 while start < len(text): end = start + CHUNK_CHARS if end < len(text): sp = text.rfind(" ", start + CHUNK_CHARS - 200, end) if sp > start: end = sp piece = text[start:end].strip() if piece: chunks.append(piece) start = end - CHUNK_OVERLAP if start < 0: start = 0 return chunks def iter_chunks(): """Genera (categoria, documento, pagina, texto) para todo el corpus.""" for root, fixed in SOURCES: if not os.path.isdir(root): print(f" [AVISO] No existe la carpeta: {root}", file=sys.stderr) continue for dirpath, _, files in os.walk(root): for fn in sorted(files): if not fn.lower().endswith(".pdf"): continue if any(x in fn.upper() for x in EXCLUDE): continue full = os.path.join(dirpath, fn) cat = category_for(root, full, fixed) doc = os.path.splitext(fn)[0] try: reader = PdfReader(full) if reader.is_encrypted: try: reader.decrypt("") except Exception: pass n_pages = len(reader.pages) except Exception as e: print(f" [SALTADO] {fn}: {e}", file=sys.stderr) continue if MAX_PAGES_PER_DOC: n_pages = min(n_pages, MAX_PAGES_PER_DOC) for pno in range(1, n_pages + 1): try: txt = reader.pages[pno - 1].extract_text() or "" except Exception: continue for piece in chunk_page(txt): yield cat, doc, pno, piece def main(): dry = "--dry" in sys.argv print("Extrayendo y fragmentando PDFs...") records = list(iter_chunks()) print(f" {len(records)} fragmentos en total.") if not records: print("No se encontraron fragmentos. Revisa las rutas en SOURCES.", file=sys.stderr) sys.exit(1) # ── Desglose ── cats_d, docs_d, total_chars = {}, {}, 0 for cat, doc, _pno, txt in records: cats_d[cat] = cats_d.get(cat, 0) + 1 docs_d[(cat, doc)] = docs_d.get((cat, doc), 0) + 1 total_chars += len(txt) print("\nFragmentos por categoria:") for c in sorted(cats_d, key=lambda k: -cats_d[k]): print(f" {c:34s} {cats_d[c]}") print(f"\nTamano estimado del index.json: ~{(total_chars + len(records) * 60) / 1e6:.1f} MB") if dry: print("\n(Modo --dry: solo diagnostico, no se escribio el indice.)") return items = [{"id": i, "cat": c, "doc": d, "page": p, "text": t} for i, (c, d, p, t) in enumerate(records)] index = {"count": len(items), "items": items} with open(OUT_FILE, "w", encoding="utf-8") as f: json.dump(index, f, ensure_ascii=False) size_mb = os.path.getsize(OUT_FILE) / 1e6 print(f"\nListo: {OUT_FILE} ({len(items)} fragmentos, {size_mb:.1f} MB)") print("Subelo al repo 'ifas_help' junto a index.html.") if __name__ == "__main__": main()
